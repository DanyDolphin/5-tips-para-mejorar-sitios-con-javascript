Interacciones con el mouse

            <script src="js/warp.js"></script>
            <script>
                const svg = document.getElementById('svg-element')
                const warp = new Warp(svg)

                warp.interpolate(6)
                warp.transform(([x, y]) => [x, y, x, y])

                let count = 100;
                let offset = 100;
                svg.addEventListener('mousemove', event => {
                    const mx = event.offsetX * 1.435,
                        my = event.offsetY * 1.435
                    warp.transform(([ x, y, ox, oy ]) => {
                        let d = Math.sqrt(Math.pow(mx - ox, 2) + Math.pow(my - oy, 2))
                        return [
                            (mx-ox) * Math.exp(d / -50) + ox, 
                            (my-oy) * Math.exp(d / -50) + oy, 
                            ox, oy
                        ]
                    })
                })
            </script>
