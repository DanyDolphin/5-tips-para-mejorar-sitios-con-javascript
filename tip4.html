            <script>
                const swiper = new Swiper('.mySwiper', {
                    speed: 400,
                    spaceBetween: 100,
                    loop: true,
                    navigation: {
                        nextEl: ".swiper-button-next",
                        prevEl: ".swiper-button-prev",
                    },
                });
            </script>

Reemplaza los componentes por default: Sweetalert2

            <div class="is-flex is-justify-content-center my-6">
                <button class="button" onclick="normalConfirm()">Alerta normal</button>
                <button class="button is-dark" onclick="sweetConfirm()">Alerta bonita</button>
            </div>
            <script src="//cdn.jsdelivr.net/npm/sweetalert2@11"></script>
            <script>
                function normalConfirm() {
                    if (confirm('Quieres proceder?')) 
                        alert('Perfecto!')
                }
    
                function sweetConfirm() {
                    Swal.fire({
                        title: 'Quieres proceder?',
                        icon: 'question',
                        showCancelButton: true
                    }).then(res => {
                        if (res.isConfirmed)
                            Swal.fire('Perfecto!','','success')
                    })
                }
            </script>
